function processCommand(command) { // Basit örnek: "bugün kaç paket A üretildi?" if (command.includes('kaç') && command.includes('paket')) { const paketMatch = command.match(/paket\s([a-z])/i); const tarihMatch = command.match(/bugün|dün|\d{4}-\d{2}-\d{2}/i); let tarih = null; if (tarihMatch) { if (tarihMatch[0] === 'bugün') { tarih = new Date().toISOString().slice(0,10); } else if (tarihMatch[0] === 'dün') { const d = new Date(); d.setDate(d.getDate() - 1); tarih = d.toISOString().slice(0,10); } else { tarih = tarihMatch[0]; } } if (paketMatch && tarih) { const paket = paketMatch[1].toUpperCase(); const toplam = productionData .filter(d => d.paket === paket && d.tarih === tarih) .reduce((sum, d) => sum + d.adet, 0); const response = ${tarih} tarihinde paket ${paket} için toplam üretim ${toplam} adettir.; output.textContent = response; speak(response); } else { const response = 'Üzgünüm, sorunu tam anlayamadım. Lütfen tekrar deneyin.'; output.textContent = response; speak(response); } } else { const response = 'Komut tanınmadı. Lütfen "kaç paket A üretildi" gibi bir soru sorunuz.'; output.textContent = response; speak(response); } }
